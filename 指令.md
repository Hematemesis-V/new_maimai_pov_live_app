不使用 NMS 导出模型时，YOLO 会吐出一个形状通常为 [1, 6, 8400] 的巨大矩阵（外键 Class 0，内屏 Class 1）。
这 8400 个锚点中，可能有几十个锚点都“看”到了内屏。但在所有这些锚点里，必然存在一个置信度达到最高峰（绝对最大值）的锚点，这个点就是模型认为最完美的内屏中心和大小。
既然内屏在画面中永远只有一个，我们根本不需要算 IoU 去抑制谁，只要在 8400 个点里，单单盯着 Class 1 的置信度，找出那个最高分，直接把它的坐标拿出来用就完事了。 这个方法不仅精准，而且极快，甚至比跑 NMS 还要省 CPU！










以下为可以参考的代码：基于实际情况使用修改,仅做参考
private func infer(_ pixelBuffer: CVPixelBuffer, preprocessMs: Double) -> DetectionResult? {
    let start = CACurrentMediaTime()

    // nms=False 的模型不需要 iouThreshold 和 confidenceThreshold
    guard let input = try? bestInput(image: pixelBuffer) else { return nil }
    guard let output = try? model.prediction(input: input) else { return nil }

    let elapsed = CACurrentMediaTime() - start
    let yoloSize = Float(Config.yoloInputSize)

    // 动态抓取输出节点，绕开 Ultralytics 随机生成的 var_1234 等奇葩名字
    let featureNames = output.featureNames
    guard let firstFeatureName = featureNames.first,
          let outputArray = output.featureValue(for: firstFeatureName)?.multiArrayValue else {
        return nil
    }

    // 检查张量形状，预期为 [1, 6, 8400] (4个坐标 + 2个类别)
    let shape = outputArray.shape
    guard shape.count == 3 else { return nil }
    
    let numFeatures = shape[1].intValue  // 应该为 6
    let numAnchors = shape[2].intValue   // 应该为 8400 或类似数值
    
    // 获取内存步长，用于一维数组指针极速跳跃
    let strides = outputArray.strides
    let strideF = strides[1].intValue // 切换 Feature 的步长
    let strideA = strides[2].intValue // 切换 Anchor 的步长

    let isFloat16 = outputArray.dataType == .float16
    let ptr32 = UnsafeMutablePointer<Float32>(OpaquePointer(outputArray.dataPointer))
    let ptr16 = UnsafeMutablePointer<Float16>(OpaquePointer(outputArray.dataPointer))

    // 🎯 核心定位：你的内屏是 Class 1。
    // YOLO 特征排列：[x, y, w, h, class0_conf, class1_conf]
    // 所以 Class 1 的索引是 5 (即 4 + 1)
    let targetClassFeatureIdx = 5 
    
    // 防御性检查：确保张量维度足够包含 Class 1
    guard numFeatures > targetClassFeatureIdx else { return nil }

    var bestConf: Float = 0.0
    var bestAnchorIdx = -1

    // ⚡️ 极速暴搜 8400 个锚点，只找 Class 1 的最高分！
    for a in 0..<numAnchors {
        let confIdx = targetClassFeatureIdx * strideF + a * strideA
        let conf: Float = isFloat16 ? Float(ptr16[confIdx]) : ptr32[confIdx]
        
        if conf > bestConf {
            bestConf = conf
            bestAnchorIdx = a
        }
    }

    let confThresh = Float(Config.defaultConfidenceThreshold)
    
    // 如果没找到，或者最高分都没过阈值，直接返回未检测到
    guard bestAnchorIdx >= 0, bestConf >= confThresh else {
        return DetectionResult(
            detected: false, confidence: bestConf,
            stabCx: 0, stabCy: 0, stabW: 0, stabH: 0,
            rawYoloCx: 0, rawYoloCy: 0, rawYoloW: 0, rawYoloH: 0,
            rawNx: 0, rawNy: 0, rawNw: 0, rawNh: 0,
            inferenceMs: elapsed * 1000.0, preprocessMs: preprocessMs,
            allBoxesCount: numAnchors, innerScreenBoxesCount: 0,
            topBoxes: "--", bestBoxRank: 0
        )
    }

    // 📦 提取最强锚点的原始坐标 (cx, cy, w, h)
    let idxCx = 0 * strideF + bestAnchorIdx * strideA
    let idxCy = 1 * strideF + bestAnchorIdx * strideA
    let idxW  = 2 * strideF + bestAnchorIdx * strideA
    let idxH  = 3 * strideF + bestAnchorIdx * strideA

    let rawNx: Float = isFloat16 ? Float(ptr16[idxCx]) : ptr32[idxCx]
    let rawNy: Float = isFloat16 ? Float(ptr16[idxCy]) : ptr32[idxCy]
    let rawNw: Float = isFloat16 ? Float(ptr16[idxW]) : ptr32[idxW]
    let rawNh: Float = isFloat16 ? Float(ptr16[idxH]) : ptr32[idxH]

    let rawCx = rawNx * yoloSize
    let rawCy = rawNy * yoloSize
    let rawW = rawNw * yoloSize
    let rawH = rawNh * yoloSize

    // 🛡️ 边界与异常值保护
    if rawW < 5.0 || rawH < 5.0 || rawCx < 0 || rawCx > yoloSize || rawCy < 0 || rawCy > yoloSize {
        return DetectionResult(
            detected: false, confidence: bestConf,
            stabCx: 0, stabCy: 0, stabW: 0, stabH: 0,
            rawYoloCx: rawCx, rawYoloCy: rawCy, rawYoloW: rawW, rawYoloH: rawH,
            rawNx: rawNx, rawNy: rawNy, rawNw: rawNw, rawNh: rawNh,
            inferenceMs: elapsed * 1000.0, preprocessMs: preprocessMs,
            allBoxesCount: numAnchors, innerScreenBoxesCount: 1,
            topBoxes: String(format: "c%.2f(err)", bestConf), bestBoxRank: 1
        )
    }

    // 📐 坐标系换算回防抖 Stab 画幅 (完全复用你之前的逻辑)
    let stabCx = (rawCx - uniforms.padLeft) / uniforms.scale - uniforms.padH
    let stabCy = (rawCy - uniforms.padTop) / uniforms.scale - uniforms.padV
    let stabW = rawW / uniforms.scale
    let stabH = rawH / uniforms.scale

    return DetectionResult(
        detected: true, confidence: bestConf,
        stabCx: stabCx, stabCy: stabCy, stabW: stabW, stabH: stabH,
        rawYoloCx: rawCx, rawYoloCy: rawCy, rawYoloW: rawW, rawYoloH: rawH,
        rawNx: rawNx, rawNy: rawNy, rawNw: rawNw, rawNh: rawNh,
        inferenceMs: elapsed * 1000.0, preprocessMs: preprocessMs,
        allBoxesCount: numAnchors, innerScreenBoxesCount: 1,
        topBoxes: String(format: "1:%.2f,%.2f,c%.2f", rawNx, rawNy, bestConf), 
        bestBoxRank: 1
    )
}
